\section{Ottimizzazione vincolata}

\begin{enumerate}
	\item
	\begin{flalign*}
		\min  z =& 4+3(1-x_{1})^{2}+(1-x_{2})^{2}\\
		\text{{s.t.}} &\\
 		& 3x_{1}+x_{2}=5
 	\end{flalign*}

	\item
	\begin{flalign*}
		\min  z =& x_{1}^{2}+4 x_{2}^{2}\\
		\text{{s.t.}} &\\
 		& x_{1}+2 x_{2} - 2 = 0
 	\end{flalign*}
 	
 	\item
	\begin{flalign*}
		\min  z =& \frac{1}{3} (x_{1} + 1)^{2}+x_{2}\\
		\text{{s.t.}}\\
 		& -x_{1} +1  \leq 0\\
 		& -x_2       \leq 0
 	\end{flalign*}
 	
\end{enumerate}
